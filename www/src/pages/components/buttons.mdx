import Button from "ui/Button.js";

# Button

## Color

```jsx react-live 
<>
<Button text="button" />
<Button color="primary" text="button" />
<Button color="blue" text="button" />
</>
```

## Size

```jsx react-live 
<>
<Button size="xs" text="Xsmall" />
<Button size="sm" text="Small" />
<Button text="Medium" />
<Button size="lg" text="Large" />
<Button width="sm" text="MinWidth Small" />
<Button width="md" text="MinWidth Medium" />
<Button width="lg" text="MinWidth Large" />
</>
```

## Disabled

```jsx react-live 
<>
<Button text="Disabled" disabled />
</>
```

## with icon


```jsx react-live 
<>
<Button title="icon" icon="ico-heloo" />
</>
```


## API
| Name     | Type    | Default | Description |
|----------|---------|---------|-------------|
| block    | boolean | false   |             |
| color    | string  |         |             |
| disabled | boolean | false   |             |
| size     | string  |         |             |
| type     |         | button  |             |
